def calculate_series(num_terms):
  """Calculates the sum of the series 1 - 1/3 + 1/5 - 1/7 + ... for a given number of terms.
  Then multiplies the result by 4."""
  total = 0
  for i in range(num_terms):
    term = 1 / (2 * i + 1)
    if i % 2 == 0:
      total += term
    else:
      total -= term
  return total * 4

num_terms = 10000
result = calculate_series(num_terms)
print(result)
